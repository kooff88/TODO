## 实例一

```html
	<div id="app">
		<ul class="persons">
			<li v-for="person in persons">name: {{ person.name }}, age: {{ person.age }};</li>
		</ul>

	</div>
	... 

	var vm = new Vue({
		el: '#app',
		data: {
			persons: [
				{
					name: 'Dale',
					age: 22
				},
				{
					name: 'Tim',
					age: 30
				},
				{
					name: 'Rex',
					age: 23
				},
			]
		}
	})

```

渲染结果:  

```html
	<div id="app">
		<ul class="persons">
			<li>name: Dale, age: 22;</li>
			<li>name: Tim, age: 30;</li>
			<li>name: Rex, age: 23;</li>
		</ul>
	</div>

```
